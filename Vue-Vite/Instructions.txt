Create Vite:
    yarn create vite .
    yarn create vite . --template vue  # with vue
    yarn create vite . --template vue-ts  # with vue and typescript 
    Now:
        yarn
        yarn dev


Add Router:
    yarn add -D vue-router
    mkdir plugins / cd _
    touch router.js
